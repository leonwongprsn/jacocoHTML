package comp3111.popnames;

/**
 * A class for store the data from csv file to a Person object.
 * @author <a href=mailto:lwwongaf@connect.ust.hk>Wong Lap Wang</a>
 * @author <a href=mailto:kytoac@connect.ust.hk>To Kai Yuen</a>
 *
 */
public class Person {
	String name;
	String gender;
	int occurrence;
	int rank;
	int yob;
	int lowest_rank;
	int highest_rank;
	int lowest_rank_year;
	int highest_rank_year;
	
	public Person(String name, String gender, int occurrence , int rank, int yob) {
		this.name = name;
		this.gender = gender;
		this.occurrence = occurrence;
		this.rank = rank;
		this.yob = yob;
		this.lowest_rank = rank;
		this.highest_rank = rank;
		this.lowest_rank_year = yob;
		this.highest_rank_year = yob;
	}
	
    public String getName() {
        return this.name;
    }
    
	
    public int getOccurrence() {
        return this.occurrence ;
    }
     
    public void addOccurrence(int increment) {
    	this.occurrence += increment;
    }
}
